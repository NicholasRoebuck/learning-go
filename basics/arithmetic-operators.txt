package main

import (
	"fmt"
	"math"
)

func main() {
	var a, b int = 10, 3

	var result  int

	result = a + b
	fmt.Println(result)

	result = a - b
	fmt.Println(result)

	result = a * b
	fmt.Println(result)

	result = a/3.0
	fmt.Println(result)

	result = a % b
	fmt.Println(result)

	const number float64 = 22/7.0
	fmt.Println(number)

	// examples of overflow with signed and unsigned int64/uint64

	var maxInt int64 = 9223372036854775807 // max int64 value
	fmt.Println(maxInt)

	maxInt += 1
	fmt.Println(maxInt) 

	var uMaxInt  uint64  = 18446744073709551615 // max value of unsigned int (uint64)
	fmt.Println(uMaxInt)

	uMaxInt += 1
	fmt.Println(uMaxInt)

	// examples of underflow 
	var smallFloat float64 = 1.0e-323
	fmt.Println(smallFloat)

	smallFloat = smallFloat / math.MaxFloat32
	fmt.Println(smallFloat)
}
